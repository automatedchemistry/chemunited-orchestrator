"""
FastAPI server for protocol configuration and pre-run inspection.

Run from anywhere — just point Python at this file:

    python path/to/project_folder/api.py

Interactive docs: http://localhost:3162/docs
"""

from __future__ import annotations

import sys
from pathlib import Path
from typing import Any

# Make the project folder importable regardless of where this script lives.
sys.path.insert(0, str(Path(__file__).parent))

from chemunited.workflow.api import RunController
from chemunited.workflow.orchestrator import Platform
from fastapi import FastAPI
from fastapi.responses import RedirectResponse

from protocols import PROCESSES  # type: ignore[import]
from protocols.main_parameters import MainParameter  # type: ignore[import]

# ── App composition ────────────────────────────────────────────────────────────

PROJECT_DIR = Path(__file__).parent
CONNECTIVITY_FILE = PROJECT_DIR / "connectivity" / "associations.json"

run = RunController(
    params=MainParameter(),
    processes=PROCESSES,
    platform=Platform.from_json(CONNECTIVITY_FILE)
    if CONNECTIVITY_FILE.exists()
    else Platform(),
    project_dir=PROJECT_DIR,
)

app = FastAPI(
    title="ChemUnited Orchestrator API",
    description="Configure and inspect a protocol before execution.",
    version="0.1.0",
)

app.include_router(run.router)


@app.get("/", include_in_schema=False)
def root() -> RedirectResponse:
    return RedirectResponse(url="/docs")


@app.get("/report", tags=["Report"], summary="Full pre-run inspection report")
def get_report() -> dict[str, Any]:
    return run.report()


# ── Entry point ────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    import uvicorn

    uvicorn.run(app, host="localhost", port=3162)
